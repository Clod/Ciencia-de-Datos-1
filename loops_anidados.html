main.py 
botellas = 3
pasos_llenado = 3
# Bucle exterior: cada botella
for i in range(botellas):
# Bucle interior: progreso de llenado
    for j in range(1, pasos_llenado + 1):
        nivel = (j / pasos_llenado) * 100
        print(f"Botella {i}: {nivel}%")
print("¡Proceso finalizado!")
